\usepackage[many]{tcolorbox}
\usepackage{graphicx}
\usetikzlibrary{calc}

% Create our custom box
\newtcolorbox{mybox}[2][]{
  enhanced,
  left=0pt,
  right=0pt,
  top=10pt,
  bottom=10pt,
  colframe = white,
  width=\textwidth,
  enlarge left by=0mm,
  boxsep=5pt,
  arc=4pt,
  outer arc=4pt,
  leftupper=1.5cm,
overlay={
    \node[anchor=west] 
      at ([xshift=10pt] $ (frame.north west)!0.5!(frame.south west) $ )
       {{\setkeys{Gin}{width=3em,keepaspectratio}\includegraphics{#2}}};},
#1}


% Define HTML colours for use
\definecolor{EFEFFF}{HTML}{EFEFFF}
\definecolor{FFFFAD}{HTML}{FFFFAD}
\definecolor{DEFFDE}{HTML}{DEFFDE}
\definecolor{FFDFDE}{HTML}{FFDFDE}


% Create the new environments for R Markdown
\newenvironment{rmdnote}
  {\begin{mybox}[colback=EFEFFF]{images/note}}
  {\end{mybox}}
  
\newenvironment{rmdimportant}
  {\begin{mybox}[colback=FFFFAD]{images/important}}
  {\end{mybox}}
  
\newenvironment{rmdtip}
  {\begin{mybox}[colback=DEFFDE]{images/tip}}
  {\end{mybox}}
  
\newenvironment{rmdwarning}
  {\begin{mybox}[colback=FFDFDE]{images/warning.png}}
  {\end{mybox}}
